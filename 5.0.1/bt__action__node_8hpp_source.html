98 void createActionClient(const std::string &action_name) {
99 // Now that we have the ROS node to use, create the action client for this
100 // BT action
101 action_client_ = rclcpp_action::create_client<ActionT>(node_, action_name,
102 callback_group_);
103
104 // Make sure the server is actually there before continuing
105 RCLCPP_DEBUG(node_->get_logger(), "Waiting for \"%s\" action server",
106 action_name.c_str());
107 if (!action_client_->wait_for_action_server(wait_for_service_timeout_)) {
108 RCLCPP_ERROR(node_->get_logger(),
109 "\"%s\" action server not available after waiting for %.2fs",
110 action_name.c_str(),
111 wait_for_service_timeout_.count() / 1000.0);
112 throw std::runtime_error(std::string("Action server ") + action_name +
113 std::string(" not available"));
114 }

188 BT::NodeStatus tick() override {
189 // first step to be done only at the beginning of the Action
190 if (!BT::isStatusActive(status())) {
191 // reset the flag to send the goal or not, allowing the user the option to
192 // set it in on_tick
193 should_send_goal_ = true;
194
195 // Clear the input and output messages to make sure we have no leftover
196 // from previous calls
197 goal_ = typename ActionT::Goal();
198 result_ =
199 typename rclcpp_action::ClientGoalHandle<ActionT>::WrappedResult();
200
201 // user defined callback, may modify "should_send_goal_".
202 on_tick();
203
204 // setting the status to RUNNING to notify the BT Loggers (if any)
205 setStatus(BT::NodeStatus::RUNNING);
206
207 if (!should_send_goal_) {
208 return BT::NodeStatus::FAILURE;
209 }
210 send_new_goal();
211 }
212
213 try {
214 // if new goal was sent and action server has not yet responded
215 // check the future goal handle
216 if (future_goal_handle_) {
217 auto elapsed = (node_->now() - time_goal_sent_)
218 .template to_chrono<std::chrono::milliseconds>();
219 if (!is_future_goal_handle_complete(elapsed)) {
220 // return RUNNING if there is still some time before timeout happens
221 if (elapsed < server_timeout_) {
222 return BT::NodeStatus::RUNNING;
223 }
224 // if server has taken more time than the specified timeout value
225 // return FAILURE
226 RCLCPP_WARN(node_->get_logger(),
227 "Timed out while waiting for action server to "
228 "acknowledge goal request for %s",
229 action_name_.c_str());
230 future_goal_handle_.reset();
231 return BT::NodeStatus::FAILURE;
232 }
233 }
234
235 // The following code corresponds to the "RUNNING" loop
236 if (rclcpp::ok() && !goal_result_available_) {
237 // user defined callback. May modify the value of "goal_updated_"
238 on_wait_for_result(feedback_);
239
240 // reset feedback to avoid stale information
241 feedback_.reset();
242
243 auto goal_status = goal_handle_->get_status();
244 if (goal_updated_ &&
245 (goal_status == action_msgs::msg::GoalStatus::STATUS_EXECUTING ||
246 goal_status == action_msgs::msg::GoalStatus::STATUS_ACCEPTED)) {
247 goal_updated_ = false;
248 send_new_goal();
249 auto elapsed = (node_->now() - time_goal_sent_)
250 .template to_chrono<std::chrono::milliseconds>();
251 if (!is_future_goal_handle_complete(elapsed)) {
252 if (elapsed < server_timeout_) {
253 return BT::NodeStatus::RUNNING;
254 }
255 RCLCPP_WARN(node_->get_logger(),
256 "Timed out while waiting for action server to "
257 "acknowledge goal request for %s",
258 action_name_.c_str());
259 future_goal_handle_.reset();
260 return BT::NodeStatus::FAILURE;
261 }
262 }
263
264 callback_group_executor_.spin_some();
265
266 // check if, after invoking spin_some(), we finally received the result
267 if (!goal_result_available_) {
268 // Yield this Action, returning RUNNING
269 return BT::NodeStatus::RUNNING;
270 }
271 }
272 } catch (const std::runtime_error &e) {
273 if (e.what() == std::string("send_goal failed") ||
274 e.what() == std::string("Goal was rejected by the action server")) {
275 // Action related failure that should not fail the tree, but the node
276 return BT::NodeStatus::FAILURE;
277 } else {
278 // Internal exception to propagate to the tree
279 throw e;
280 }
281 }
282
283 BT::NodeStatus status;
284 switch (result_.code) {
285 case rclcpp_action::ResultCode::SUCCEEDED:
286 status = on_success();
287 break;
288
289 case rclcpp_action::ResultCode::ABORTED:
290 status = on_aborted();
291 break;
292
293 case rclcpp_action::ResultCode::CANCELED:
294 status = on_cancelled();
295 break;
296
297 default:
298 throw std::logic_error("BtActionNode::Tick: invalid status value");
299 }
300
301 goal_handle_.reset();
302 return status;
303 }

309 void halt() override {
310 if (should_cancel_goal()) {
311 auto future_result = action_client_->async_get_result(goal_handle_);
312 auto future_cancel = action_client_->async_cancel_goal(goal_handle_);
313 if (callback_group_executor_.spin_until_future_complete(
314 future_cancel, server_timeout_) !=
315 rclcpp::FutureReturnCode::SUCCESS) {
316 RCLCPP_ERROR(node_->get_logger(),
317 "Failed to cancel action server for %s",
318 action_name_.c_str());
319 }
320
321 if (callback_group_executor_.spin_until_future_complete(
322 future_result, server_timeout_) !=
323 rclcpp::FutureReturnCode::SUCCESS) {
324 RCLCPP_ERROR(node_->get_logger(),
325 "Failed to get result for %s in node halt!",
326 action_name_.c_str());
327 }
328
329 on_cancelled();
330 }
331
332 // this is probably redundant, since the parent node
333 // is supposed to call it, but we keep it, just in case
334 resetStatus();
335 }

342 bool should_cancel_goal() {
343 // Shut the node down if it is currently running
344 if (status() != BT::NodeStatus::RUNNING) {
345 return false;
346 }
347
348 // No need to cancel the goal if goal handle is invalid
349 if (!goal_handle_) {
350 return false;
351 }
352
353 callback_group_executor_.spin_some();
354 auto status = goal_handle_->get_status();
355
356 // Check if the goal is still executing
357 return status == action_msgs::msg::GoalStatus::STATUS_ACCEPTED ||
358 status == action_msgs::msg::GoalStatus::STATUS_EXECUTING;
359 }

364 void send_new_goal() {
365 goal_result_available_ = false;
366 auto send_goal_options =
367 typename rclcpp_action::Client<ActionT>::SendGoalOptions();
368 send_goal_options.result_callback =
369 [this](const typename rclcpp_action::ClientGoalHandle<
370 ActionT>::WrappedResult &result) {
371 if (future_goal_handle_) {
372 RCLCPP_DEBUG(
373 node_->get_logger(),
374 "Goal result for %s available, but it hasn't received the goal "
375 "response yet. "
376 "It's probably a goal result for the last goal request",
377 action_name_.c_str());
378 return;
379 }
380
381 // TODO(#1652): a work around until rcl_action interface is updated
382 // if goal ids are not matched, the older goal call this callback so
383 // ignore the result if matched, it must be processed (including
384 // aborted)
385 if (this->goal_handle_->get_goal_id() == result.goal_id) {
386 goal_result_available_ = true;
387 result_ = result;
388 emitWakeUpSignal();
389 }
390 };
391 send_goal_options.feedback_callback =
392 [this](
393 typename rclcpp_action::ClientGoalHandle<ActionT>::SharedPtr,
394 const std::shared_ptr<const typename ActionT::Feedback> feedback) {
395 feedback_ = feedback;
396 emitWakeUpSignal();
397 };
398
399 future_goal_handle_ = std::make_shared<std::shared_future<
400 typename rclcpp_action::ClientGoalHandle<ActionT>::SharedPtr>>(
401 action_client_->async_send_goal(goal_, send_goal_options));
402 time_goal_sent_ = node_->now();
403 }

413 bool is_future_goal_handle_complete(std::chrono::milliseconds &elapsed) {
414 auto remaining = server_timeout_ - elapsed;
415
416 // server has already timed out, no need to sleep
417 if (remaining <= std::chrono::milliseconds(0)) {
418 future_goal_handle_.reset();
419 return false;
420 }
421
422 auto timeout = remaining > max_timeout_ ? max_timeout_ : remaining;
423 auto result = callback_group_executor_.spin_until_future_complete(
424 *future_goal_handle_, timeout);
425 elapsed += timeout;
426
427 if (result == rclcpp::FutureReturnCode::INTERRUPTED) {
428 future_goal_handle_.reset();
429 throw std::runtime_error("send_goal failed");
430 }
431
432 if (result == rclcpp::FutureReturnCode::SUCCESS) {
433 goal_handle_ = future_goal_handle_->get();
434 future_goal_handle_.reset();
435 if (!goal_handle_) {
436 throw std::runtime_error("Goal was rejected by the action server");
437 }
438 return true;
439 }
440
441 return false;
442 }
