167static inline std::string random_string(int string_size) {
168 static const std::string str(
169 "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz");
170
171 std::random_device rd;
172 std::mt19937 generator(rd());
173
174 std::string result(string_size, ' ');
175
176 for (int i = 0; i < string_size; ++i) {
177 result[i] = str[generator() % str.size()];
178 }
179
180 return result;
181}
