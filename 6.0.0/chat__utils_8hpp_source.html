56static inline std::string random_string(int string_size) {
57 static const std::string str(
58 "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz");
59
60 std::random_device rd;
61 std::mt19937 generator(rd());
62
63 std::string result(string_size, ' ');
64
65 for (int i = 0; i < string_size; ++i) {
66 result[i] = str[generator() % str.size()];
67 }
68
69 return result;
70}

82inline size_t validate_utf8(const std::string &text) {
83 size_t len = text.size();
84 if (len == 0)
85 return 0;
86
87 // Check the last few bytes to see if a multi-byte character is cut off
88 for (size_t i = 1; i <= 4 && i <= len; ++i) {
89 unsigned char c = text[len - i];
90 // Check for start of a multi-byte sequence from the end
91 if ((c & 0xE0) == 0xC0) {
92 // 2-byte character start: 110xxxxx
93 // Needs at least 2 bytes
94 if (i < 2)
95 return len - i;
96 } else if ((c & 0xF0) == 0xE0) {
97 // 3-byte character start: 1110xxxx
98 // Needs at least 3 bytes
99 if (i < 3)
100 return len - i;
101 } else if ((c & 0xF8) == 0xF0) {
102 // 4-byte character start: 11110xxx
103 // Needs at least 4 bytes
104 if (i < 4)
105 return len - i;
106 }
107 }
108
109 // If no cut-off multi-byte character is found, return full length
110 return len;
111}
