163inline std::string random_string(int string_size = 32) {
164 static const std::string str(
165 "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz");
166
167 std::random_device rd;
168 std::mt19937 generator(rd());
169
170 std::string result(string_size, ' ');
171
172 for (int i = 0; i < string_size; ++i) {
173 result[i] = str[generator() % str.size()];
174 }
175
176 return result;
177}
