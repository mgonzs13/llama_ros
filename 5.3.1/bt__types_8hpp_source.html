67template <>
68inline llama_msgs::msg::Response convertFromString(BT::StringView str) {
69
70#if not defined(BTV3)
71 if (StartWith(str, "json:")) {
72 str.remove_prefix(5);
73 return convertFromJSON<llama_msgs::msg::Response>(str);
74 }
75#endif
76
77 llama_msgs::msg::Response output;
78 if (!str.empty()) {
79 // We expect values separated by /
80 auto parts = splitString(str, '/');
81 if (parts.size() != 2) {
82 throw RuntimeError("invalid input)");
83 } else {
84 output.text = convertFromString<std::string>(parts[0]);
85 output.tokens = convertFromString<std::vector<int>>(parts[1]);
86 }
87 }
88 return output;
89}

100template <>
101inline std::vector<std::string> convertFromString(BT::StringView str) {
102 auto parts = splitString(str, ';');
103 std::vector<std::string> output;
104 output.reserve(parts.size());
105 for (const StringView &part : parts) {
106 output.push_back(convertFromString<std::string>(part));
107 }
108 return output;
109}

122template <>
123inline std::vector<llama_msgs::msg::ChatMessage>
124convertFromString<std::vector<llama_msgs::msg::ChatMessage>>(
125 BT::StringView str) {
126 fprintf(stderr, "Parsing chat message: %s\n", str.data());
127 std::vector<llama_msgs::msg::ChatMessage> output;
128 if (!str.empty()) {
129 auto data = nlohmann::json::parse(str.data());
130
131 for (size_t i = 0; i < data.size(); i++) {
132 auto message = data[i];
133 llama_msgs::msg::ChatMessage chat_message;
134 chat_message.role = data["role"];
135 if (data.contains("content")) {
136 chat_message.content = data["content"];
137 }
138 if (data.contains("reasoning_content")) {
139 chat_message.reasoning_content = data["reasoning_content"];
140 }
141 if (data.contains("tool_name")) {
142 chat_message.tool_name = data["tool_name"];
143 }
144 if (data.contains("tool_call_id")) {
145 chat_message.tool_call_id = data["tool_call_id"];
146 }
147 if (data.contains("content_parts")) {
148 std::vector<llama_msgs::msg::ChatContent> content_parts;
149 for (const auto &part : data["content_parts"]) {
150 llama_msgs::msg::ChatContent content_part;
151 content_part.type = part["type"];
152 content_part.text = part["text"];
153 content_parts.push_back(content_part);
154 }
155 chat_message.content_parts = content_parts;
156 }
157 if (data.contains("tool_calls")) {
158 std::vector<llama_msgs::msg::ChatToolCall> tool_calls;
159 for (const auto &tool_call : data["tool_calls"]) {
160 llama_msgs::msg::ChatToolCall tool_call_msg;
161 tool_call_msg.name = tool_call["name"];
162 tool_call_msg.id = tool_call["id"];
163 tool_call_msg.arguments = tool_call["arguments"];
164 tool_calls.push_back(tool_call_msg);
165 }
166 chat_message.tool_calls = tool_calls;
167 }
168 output.push_back(chat_message);
169 }
170 }
171 return output;
172}

185template <>
186inline std::vector<llama_msgs::msg::ChatReqTool>
187convertFromString<std::vector<llama_msgs::msg::ChatReqTool>>(
188 BT::StringView str) {
189 std::vector<llama_msgs::msg::ChatReqTool> output;
190 if (!str.empty()) {
191 auto data = nlohmann::json::parse(str.data());
192
193 for (size_t i = 0; i < data.size(); i++) {
194 auto tool = data[i];
195 llama_msgs::msg::ChatReqTool chat_req_tool;
196
197 if (data.contains("function")) {
198 llama_msgs::msg::ChatTool function;
199 function.name = data["function"]["name"];
200 function.description = data["function"]["description"];
201 function.parameters = data["function"]["parameters"];
202 chat_req_tool.function = function;
203 }
204
205 output.push_back(chat_req_tool);
206 }
207 }
208 return output;
209}
