172static inline std::string random_string(int string_size) {
173 static const std::string str(
174 "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz");
175
176 std::random_device rd;
177 std::mt19937 generator(rd());
178
179 std::string result(string_size, ' ');
180
181 for (int i = 0; i < string_size; ++i) {
182 result[i] = str[generator() % str.size()];
183 }
184
185 return result;
186}
